<# .SYNOPSIS Build web game and deploy to MuzaMSK/dist for GitHub Pages Usage: .\deploy-to-dist.ps1 -Message "Update v1.3" #> param( [string]$Message = "Deploy: $(Get-Date -Format 'yyyy-MM-dd HH:mm')" ) Write-Host "Building and deploying to MuzaMSK/dist..." -ForegroundColor Cyan # === CONFIG === $SourceRoot = "$PSScriptRoot" $BuildFolder = "$SourceRoot/build" $DistRepo = "$SourceRoot/../dist" # ============== # 1. Build the web game Write-Host "Building WebAssembly..." -ForegroundColor Yellow & "$SourceRoot/build.ps1" -Target web if ($LASTEXITCODE -ne 0) { Write-Host "Build failed!" -ForegroundColor Red exit 1 } # 2. Verify dist repo exists if (!(Test-Path $DistRepo)) { Write-Host "Dist repo not found at: $DistRepo" -ForegroundColor Red Write-Host "Clone it: git clone https://github.com/MuzaMSK/dist ../dist" -ForegroundColor Yellow exit 1 } # 3. Clean dist repo (preserve .git and essential files) Write-Host "Cleaning dist repo..." -ForegroundColor Yellow Get-ChildItem -Path $DistRepo -Exclude ".git", ".gitignore", "README.md", "CNAME" | Remove-Item -Recurse -Force -ErrorAction SilentlyContinue # 4. Copy build output to dist Write-Host "Copying build files..." -ForegroundColor Yellow Copy-Item "$BuildFolder/*" -Destination $DistRepo -Recurse -Force # 5. Commit and push Write-Host "Pushing to GitHub..." -ForegroundColor Yellow Set-Location $DistRepo git add . git commit -m $Message git push origin main if ($LASTEXITCODE -eq 0) { Write-Host "Deploy successful!" -ForegroundColor Green Write-Host "Live at: https://muzamsk.github.io/dist/" -ForegroundColor Cyan } else { Write-Host "Push failed!" -ForegroundColor Red exit 1 } # Return to source Set-Location $SourceRoot
